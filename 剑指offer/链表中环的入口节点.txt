import java.util.*;

public class Solution {

    public ListNode EntryNodeOfLoop(ListNode pHead)
        {
        Map newMap = new HashMap();
        ListNode p = pHead;
        int i = 0;
        while(p!=null){
            if(newMap.containsValue(p)){
                return p;
            }
            newMap.put(i,p);
            i++;
        p=p.next;
        }
        return null;
    }
}
