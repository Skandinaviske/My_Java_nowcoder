public class Solution {
    public TreeLinkNode GetNext(TreeLinkNode pNode)
        {
        if(pNode.right==null){
            if(pNode.next==null)return null;
            if(pNode.next.left == pNode)return pNode.next;
            while(pNode.next!=null){
                if(pNode.next.left==pNode)return pNode.next;
                pNode = pNode.next;
            }
            return null;
        }
        else
        {
            pNode =pNode.right;
            while(pNode.left!=null){
            pNode = pNode.left;
        }
        return pNode;
        }
        //return null;
    }
}
